select control rendered as list control

This form control allows the user to make multiple selections from a set of choices.

A select1 with appearance 'compact' will be rendered as a list conrol.

XForms Markup

<xf:select1 id="select1" ref="select1/value" appearance="compact" incremental="true">
    <xf:label>a select1 list</xf:label>
    <xf:hint>a Hint for this control</xf:hint>
    <xf:help>help for select1</xf:help>
    <xf:alert>invalid</xf:alert>
    <xf:itemset nodeset="instance('codes')/country">
        <xf:label ref="country-name"></xf:label>
        <xf:value ref="code"></xf:value>
    </xf:itemset>
</xf:select1>
